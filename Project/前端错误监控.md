# 前端错误监控策略

1. JS 错误捕获：
   - 使用 `window.addEventListener('error',...)` 捕获同步运行时错误
   - 使用 `window.addEventListener('onunhandledrejection',...)` 捕获未处理的 Promise 异常

2. 网络请求监控：
   - 通过重写 `XMLHttpRequest.prototype.open` 和 `window.fetch` 来拦截接口报错，要排除 SDK 自身的上报请求，防止陷入“报错-上报-报错”的死循环

3. 资源加载监控：
   - 利用 `window.addEventListener('error', handler, true)` 在**捕获阶段**拦截 `<img>`、`<script>` 等标签的加载失败

4. 框架深度集成：
   - Vue: 需接入 `Vue.config.errorHandler`，否则很多组件内部错误会被 `Vue` 吞掉，不会触发全局 `onerror`
   - React: 需利用 `ErrorBoundary` 捕获组件树崩溃

5. 白屏监控：
   - 通过 `MutationObserver` 或检测根节点是否有 DOM 渲染来判断是否发生白屏

6. 卡顿监控：
   - 通过 `requestAnimationFrame` 监测长任务

7. 错误数据上报策略：
   - 优先使用 `navigator.sendBeacon`，确保页面卸载时数据也能成功发出
   - 兜底使用 `fetch` 或 `XHR`

8. SourceMap 反解：
   - 生产环境代码是压缩混淆过的，只上报堆栈行号（如 `app.js:1:500`）毫无意义。后端需要建立一套根据 `SourceMap` 自动还原原始代码行号的机制

9. 用户行为回溯：
   - 仅仅看报错行是不够的。你需要记录报错前用户做了什么（点击了哪个按钮、跳转了哪个路由、之前的接口返回了什么），将这些行为随同错误一同上报，才能实现快速复现

错误日志交给`worker`处理：

```ts
// monitor.worker.js
const state = {
  reportUrl: "",
  queue: [],
  errorCache: new Set(),
  sampleRate: 0.2,
}

// 1. 简单 Hash 生成函数
function generateHash(msg, stack) {
  let str = `${msg}${stack}`
  let hash = 0
  for (let i = 0; i < str.length; i++) {
    hash = (hash << 5) - hash + str.charCodeAt(i)
    hash |= 0
  }
  return hash.toString()
}

// 2. 执行上报
function flush() {
  if (state.queue.length === 0) return
  const data = JSON.stringify(state.queue)
  state.queue = []

  // Worker 中没有 navigator.sendBeacon，使用 fetch
  // 但可以利用 keepalive 属性达到类似效果
  fetch(state.reportUrl, {
    method: "POST",
    body: data,
    headers: { "Content-Type": "application/json" },
    keepalive: true,
  }).catch(() => {})
}

// 3. 监听主线程消息
self.onmessage = function (e) {
  const { type, payload, config } = e.data

  if (type === "INIT") {
    state.reportUrl = config.reportUrl
    state.sampleRate = config.sampleRate
    // 每 5 秒自动检查一次队列
    setInterval(flush, 5000)
    return
  }

  if (type === "REPORT") {
    // 采样拦截
    if (payload.type === "perf" && Math.random() > state.sampleRate) return

    // 去重逻辑
    const hash = generateHash(payload.message, payload.stack || "")
    if (state.errorCache.has(hash)) return

    state.errorCache.add(hash)
    setTimeout(() => state.errorCache.delete(hash), 60000)

    state.queue.push(payload)
    if (state.queue.length >= 10) flush()
  }

  if (type === "FLUSH") flush()
}
```

`MonitorSDK`类：

```ts
class MonitorSDK {
  constructor(options) {
    this.options = options
    this.breadcrumbs = []
    // 1. 初始化 Worker
    this.worker = new Worker(new URL("./monitor.worker.js", import.meta.url))
    this.worker.postMessage({ type: "INIT", config: options })

    this.init()
  }

  // 统一派发给 Worker 处理
  postToWorker(data) {
    this.worker.postMessage({
      type: "REPORT",
      payload: {
        ...data,
        project: this.options.project,
        url: location.href,
        ua: navigator.userAgent,
        breadcrumbs: [...this.breadcrumbs],
        timestamp: Date.now(),
      },
    })
  }

  init() {
    this.captureJsError()
    this.capturePromiseError()
    this.captureResourceError()
    this.captureNetwork()
    this.captureWhiteScreen()
    this.captureBehavior()
    this.capturePerformance()

    // 页面关闭前通知 Worker 强制清空
    window.addEventListener("beforeunload", () => {
      this.worker.postMessage({ type: "FLUSH" })
    })
  }

  // --- 采集器实现 ---
  // js错误
  captureJsError() {
    window.addEventListener(
      "error",
      (e) => {
        if (e.target.localName) return // 过滤资源
        this.postToWorker({ type: "js", message: e.message, stack: e.error?.stack })
      },
      true,
    )
  }
  // promise错误
  capturePromiseError() {
    window.addEventListener("unhandledrejection", (e) => {
      this.postToWorker({ type: "promise", message: e.reason?.message || String(e.reason) })
    })
  }
  // 资源引用错误-第三个参数true（因为img/script之类不会在冒泡阶段捕获）
  captureResourceError() {
    window.addEventListener(
      "error",
      (e) => {
        const target = e.target
        if (target.src || target.href) {
          this.postToWorker({
            type: "resource",
            message: `Load failed: ${target.src || target.href}`,
            meta: { tagName: target.tagName },
          })
        }
      },
      true,
    )
  }
  // 网络错误-拦截fetch
  captureNetwork() {
    const originalFetch = window.fetch
    window.fetch = async (...args) => {
      const url = args[0] instanceof Request ? args[0].url : args[0]
      if (url.includes(this.options.reportUrl)) return originalFetch(...args)

      try {
        const response = await originalFetch(...args)
        if (!response.ok) {
          this.postToWorker({ type: "ajax", message: `HTTP ${response.status}`, meta: { url } })
        }
        return response
      } catch (err) {
        this.postToWorker({
          type: "ajax",
          message: "Fetch Failed",
          meta: { url, err: err.message },
        })
        throw err
      }
    }
  }
  // 行为采集
  captureBehavior() {
    window.addEventListener(
      "click",
      (e) => {
        const t = e.target
        this.breadcrumbs.push({ t: "click", v: `${t.tagName}#${t.id}`, time: Date.now() })
        if (this.breadcrumbs.length > 10) this.breadcrumbs.shift()
      },
      true,
    )
  }
  // 性能采集
  capturePerformance() {
    if (window.PerformanceObserver) {
      const obs = new PerformanceObserver((list) => {
        list.getEntries().forEach((entry) => {
          if (entry.duration > 100) {
            this.postToWorker({
              type: "perf",
              message: "Long Task",
              meta: { duration: entry.duration },
            })
          }
        })
      })
      obs.observe({ entryTypes: ["longtask"] })
    }
  }
  // 白屏监测
  captureWhiteScreen() {
    // 确保在页面加载完成（如 onload）后的 2-3 秒进行检查
    window.addEventListener("load", () => {
      let skeletonStrategy = 0 // 这里的策略可以根据业务调整
      const MAX_RETRY = 3 // 采样重试次数
      let retryCount = 0

      const check = () => {
        const root = document.getElementById("root") || document.body
        // 如果根节点内没有子元素，或者只有注释/空白
        if (this.isContainerEmpty(root)) {
          retryCount++
          if (retryCount < MAX_RETRY) {
            setTimeout(check, 1000) // 延迟重试，排除加载极慢的情况
          } else {
            this.postToWorker({
              type: "perf",
              message: "White Screen Detected",
              meta: {
                selector: "#root",
                retry: retryCount,
                domNodes: root.childNodes.length,
              },
            })
          }
        }
      }

      // 也可以利用 requestIdleCallback 在浏览器空闲时检查
      if ("requestIdleCallback" in window) {
        window.requestIdleCallback(check)
      } else {
        setTimeout(check, 3000)
      }
    })
  }

  // 判断容器是否真的“空”了
  isContainerEmpty(element) {
    if (!element) return true
    // 排除掉只有空白文本或脚本标签的情况
    const validNodes = Array.from(element.childNodes).filter((node) => {
      return node.nodeType === 1 || (node.nodeType === 3 && node.textContent.trim() !== "")
    })
    return validNodes.length === 0
  }

  // 框架集成接口
  logFrameworkError(err, info) {
    this.postToWorker({ type: "framework", message: err.message, stack: err.stack, meta: { info } })
  }
}

export default MonitorSDK
```

实例化`MonitorSDK`并开始监控上报：

```ts
// -----创建MonitorSDK实例-----
import MonitorSDK from "./utils/MonitorSDK.js"

// 1. 全局初始化
const monitor = new MonitorSDK({
  project: "xxx", // 项目标识
  reportUrl: "https://log-server.com/api/v1/report", // 后端接收地址
  env: "production", // 当前环境
  sampleRate: 0.1, // 性能数据采样率为 10%
})

// 2. 将实例挂载到全局（可选，方便在业务代码中手动打点）
window.monitor = monitor

// -----结合Vue使用-----
import { createApp } from "vue"
import App from "./App.vue"

const app = createApp(App)

// 接入 Vue 错误处理
app.config.errorHandler = (err, instance, info) => {
  // 调用 SDK 暴露的框架集成接口
  monitor.logFrameworkError(err, info)

  // 依然在控制台打印错误，方便开发调试
  console.error(err)
}

app.mount("#app")

// -----结合React使用-----
// hooks/useMonitor.js
import { useEffect } from "react"
import { monitor } from "../utils/monitor"

export function useComponentMonitor(componentName) {
  useEffect(() => {
    const startTime = performance.now()

    return () => {
      const duration = performance.now() - startTime
      // 上报组件卸载时的停留时长（采样）
      monitor.postToWorker({
        type: "perf",
        message: `Component ${componentName} persistence`,
        meta: { duration },
      })
    }
  }, [componentName])
}

// 在根节点包裹
// <GlobalErrorBoundary><App /></GlobalErrorBoundary>

// -----业务代码中结合实际情况使用使用-----
async function handlePayment() {
  try {
    const res = await api.pay()
    if (res.code !== 200) {
      // 虽然不是 JS 运行时错误，但需要记录业务异常
      window.$monitor.postToWorker({
        type: "business",
        message: "Payment Failed",
        meta: { userId: 123, orderId: "ABC", reason: res.msg },
      })
    }
  } catch (e) {
    // 异步异常手动捕获
    window.$monitor.postToWorker({ type: "js", message: e.message, stack: e.stack })
  }
}
```
